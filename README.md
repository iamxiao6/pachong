def get_one_page(url):
    try:
        response = requests.get(url)#拿到网页数据
        if response.status_code == 200:#返回200表示响应正常
            return response.text#返回数据
        return None#如果响应不正常，则不返回任何数据
    except RequestException as e:#所有异常输出为空

        return None
